# Diplomat

beginner_diplomat = {
	type = skill
	texture = "gfx/interface/icons/character_trait_icons/beginner_diplomat.dds"
	
	character_modifier = {
		character_popularity_add = 10
	}
	
	command_modifier = {
		unit_morale_recovery_mult = 0.25
	}

	country_modifier = {
		country_influence_mult = 0.05
	}

	possible = {
		NOR = {
			has_trait = intermediate_diplomat
			has_trait = advanced_diplomat
			has_trait = expert_diplomat
		}
	}
	
	weight = {
		value = 5
		if = {
			limit = {
				has_role = politician
			}
			multiply = 4
		}
		if = {
			limit = {
				OR = {
					has_trait = tactful
					has_trait = charismatic
					has_trait = direct
					has_trait = elawyer
				}
			}
			multiply = 2
		}
		if = {
			limit = {
				OR = {
					has_trait = imperious
					has_trait = wrathful
					has_trait = reckless
					has_trait = dramatic
				}
			}
			divide = 4
		}
	}
	
	replace = {
	}
	
	value = 1
}
